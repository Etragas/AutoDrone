              #!/usr/bin/env python
              #import roslib

              import rospy
              from geometry_msgs.msg import Twist
              from std_msgs.msg import Empty
              from ardrone_autonomy.msg import Navdata
              import pid
              import time

              class ArdroneControl:
              def __init__( self ):
              self.nav_sub = rospy.Subscriber( "ardrone/navdata", Navdata, self.callback_navdata )
              self.cmd_vel_pub = rospy.Publisher( "cmd_vel", Twist, queue_size=1 )
              self.cmd_takeoff = rospy.Publisher( "ardrone/takeoff", Empty, queue_size=1 )
              self.cmd_land = rospy.Publisher( "ardrone/land", Empty, queue_size=1 )
              self.goal_vel_sub = rospy.Subscriber( "goal_vel", Twist, self.callback_goal_vel )

              def callback_navdata( self, data ):
              self.vx = data.vx/1e3
              self.vy = data.vy/1e3
              self.vz = data.vz/1e3

              print("raw navdata is vx {} vy {} vz {} ".format(data.vx,data.vy,data.vz))
              print("navdata is vx {} vy {} vz {} ".format(self.vx,self.vy,self.vz))

              def start( self ):
              cmd = Empty()
              self.cmd_takeoff.publish( cmd )

              def stop( self ):
              cmd = Empty()
              self.cmd_land.publish( cmd )

              def main():
              rospy.init_node( 'ardrone_control' )

              control = ArdroneControl()
              r = rospy.Rate(1)

              try:
              while not rospy.is_shutdown():
              control.start()
              print "sleep 1"
              except KeyboardInterrupt:
              print "Shutting down"

              if __name__ == '__main__':
              main()

                                                              Mike Hamer Shafiqladha1990 • 3 years ago

                                                              So it seems that the Dualshock 3 is poorly supported by Linux. However because it is often used to control robots, Willow Garage (the creators of ROS) have written a driver!

                                                              Unfortunately I cannot test it for you as I don't have a Dualshock 3, however try to follow these instructions:
                                                              http://www.ros.org/wiki/ps3joy...

                                                              You may run into difficulties with the bluetooth not being available in the virtual machine. To enable this, take a look in the virtual machine settings under the Ports / USB tab (where I instruct you to install the joystick). Here you should hopefully see a bluetooth device which you can then add to the virtual machine.

                                                              Once you've added the bluetooth device, any communication (eg with the gamepad) will go straight to the virtual machine to be handled by the ROS driver.

                                                              Hopefully this works for you! If you encounter any problems, write back and I'll try as best I can to help :-)
